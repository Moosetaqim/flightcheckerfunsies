        function initDB() {
            return new Promise((resolve, reject) => {
                const request = indexedDB.open('SATSFlightDB', 1);
                
                request.onerror = () => reject(request.error);
                request.onsuccess = () => {
                    db = request.result;
                    resolve(db);
                };
                
                request.onupgradeneeded = (event) => {
                    const db = event.target.result;
                    if (!db.objectStoreNames.contains('flights')) {
                        const objectStore = db.createObjectStore('flights', { autoIncrement: true });
                        objectStore.createIndex('airline', 'Airline', { unique: false });
                    }
                };
            });
        }
        
        function populateDB() {
            return new Promise((resolve, reject) => {
                const transaction = db.transaction(['flights'], 'readwrite');
                const objectStore = transaction.objectStore('flights');
                
                const clearRequest = objectStore.clear();
                
                clearRequest.onsuccess = () => {
                    FLIGHT_DATA.forEach(flight => {
                        objectStore.add(flight);
                    });
                };
                
                transaction.oncomplete = () => resolve();
                transaction.onerror = () => reject(transaction.error);
            });
        }
        
        function loadFlights() {
            return new Promise((resolve, reject) => {
                const transaction = db.transaction(['flights'], 'readonly');
                const objectStore = transaction.objectStore('flights');
                const request = objectStore.getAll();
                
                request.onsuccess = () => resolve(request.result);
                request.onerror = () => reject(request.error);
            });
        }
        
        async function initializeApp() {
            try {
                await initDB();
                await populateDB();
                flightData = await loadFlights();
                
                displayResults();
                updateResultsTitle();
            } catch (error) {
                console.error('Database error:', error);
                flightData = FLIGHT_DATA;
                displayResults();
            }
        }
        
        function initAudio() {
            try {
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                if (!audioContext) {
                    audioContext = new AudioContext();
                }
                if (audioContext.state === 'suspended') {
                    audioContext.resume();
                }
                return true;
            } catch (e) {
                console.error('Audio init failed:', e);
                return false;
            }
        }
        
        document.getElementById('searchQuery').addEventListener('input', debounce(displayResults, 300));
        document.getElementById('daySearch').addEventListener('input', debounce(displayResults, 300));
        
        function debounce(func, wait) {
            let timeout;
            return function(...args) {
                clearTimeout(timeout);
                timeout = setTimeout(() => func(...args), wait);
            };
        }
        
        function displayResults() {
            if (activeTimer !== null) return;
            
            const searchQuery = document.getElementById('searchQuery').value.toLowerCase();
            const daySearch = document.getElementById('daySearch').value.toLowerCase();
            
            const filtered = flightData.filter(flight => {
                const searchMatch = !searchQuery || 
                    (flight['Arr Flt'] && String(flight['Arr Flt']).toLowerCase().includes(searchQuery)) ||
                    (flight['Dep Flt'] && String(flight['Dep Flt']).toLowerCase().includes(searchQuery)) ||
                    (flight['Airline Name'] && String(flight['Airline Name']).toLowerCase().includes(searchQuery)) ||
                    (flight['Airline'] && String(flight['Airline']).toLowerCase().includes(searchQuery));
                
                const dayMatch = !daySearch || 
                    (flight['Day'] && String(flight['Day']).toLowerCase().includes(daySearch)) ||
                    (flight['Frequency'] && String(flight['Frequency']).toLowerCase().includes(daySearch));
                
                return searchMatch && dayMatch;
            });
            
            const resultsContainer = document.getElementById('results');
            
            if (filtered.length === 0) {
                resultsContainer.innerHTML = '<div class="no-results"><p>No flights found matching your search criteria.</p></div>';
            } else {
                resultsContainer.innerHTML = filtered.map((flight, index) => createFlightCard(flight, index)).join('');
            }
            
            updateResultsTitle(filtered.length);
            toggleClearButton(searchQuery || daySearch);
        }
        
        function createFlightCard(flight, index) {
            const arrFlt = flight['Arr Flt'] || '';
            const depFlt = flight['Dep Flt'] || '';
            const airlineCode = flight['Airline'] || '';
            const airlineName = flight['Airline Name'] || '';
            const sta = flight['STA'] || '';
            const std = flight['STD'] || '';
            const day = flight['Day'] || '';
            const precisionTime = flight['Precision Time'] || 0;
            const route = flight['Route'] || '';
            
            const primaryFlightNumber = arrFlt || depFlt || 'N/A';
            const correspondingFlightNumber = arrFlt && depFlt ? (arrFlt ? depFlt : arrFlt) : '';
            
            return `
                <div class="flight-card" id="flight-${index}">
                    <div class="flight-header">
                        <div>
                            <div class="flight-number">${primaryFlightNumber}</div>
                            <div class="airline-name">${airlineCode}${airlineName ? ' - ' + airlineName : ''}</div>
                        </div>
                        ${correspondingFlightNumber ? `<span class="badge-secondary">Corresponding: ${correspondingFlightNumber}</span>` : ''}
                    </div>
                    <div class="flight-details">
                        ${sta ? `<div class="detail-item"><strong>STA:</strong> ${sta}</div>` : ''}
                        ${std ? `<div class="detail-item"><strong>STD:</strong> ${std}</div>` : ''}
                        ${day ? `<div class="detail-item"><strong>Days:</strong> ${day}</div>` : ''}
                        ${route ? `<div class="detail-item"><strong>Route:</strong> ${route}</div>` : ''}
                    </div>
                    ${precisionTime > 0 ? `
                        <div class="timer-section">
                            <div class="timer-display" id="timer-${index}">Ready to start</div>
                            <button class="timer-btn start-timer-btn" id="start-btn-${index}" onclick="startTimer(${index}, ${precisionTime})">Start ${precisionTime} Min Timer</button>
                            <button class="timer-btn stop-timer-btn" id="stop-btn-${index}" onclick="stopTimer(${index})" style="display: none;">Stop Timer</button>
                        </div>
                    ` : ''}
                </div>
            `;
        }
        
        function startTimer(flightIndex, minutes) {
            if (activeTimer !== null) {
                alert('A timer is already running!');
                return;
            }
            
            initAudio();
            activeTimer = flightIndex;
            const warningTime = 2 * 60;
            const endTime = Date.now() + (minutes * 60 * 1000);
            
            const card = document.getElementById(`flight-${flightIndex}`);
            const timerDisplay = document.getElementById(`timer-${flightIndex}`);
            const startBtn = document.getElementById(`start-btn-${flightIndex}`);
            const stopBtn = document.getElementById(`stop-btn-${flightIndex}`);
            
            document.querySelectorAll('.flight-card').forEach((c, idx) => {
                c.style.display = idx === flightIndex ? 'block' : 'none';
            });
            
            card.classList.add('timer-active');
            startBtn.style.display = 'none';
            stopBtn.style.display = 'block';
            card.dataset.endTime = endTime;
            card.dataset.warningTriggered = 'false';
            
            const interval = setInterval(() => {
                const remainingMs = endTime - Date.now();
                const totalSeconds = Math.max(0, Math.floor(remainingMs / 1000));
                const mins = Math.floor(totalSeconds / 60);
                const secs = totalSeconds % 60;
                timerDisplay.textContent = `${mins}:${secs.toString().padStart(2, '0')} remaining`;
                
                if (totalSeconds <= warningTime && card.dataset.warningTriggered === 'false') {
                    card.classList.add('timer-warning');
                    timerDisplay.classList.add('warning');
                    card.dataset.warningTriggered = 'true';
                    playWarning();
                }
                
                if (remainingMs <= 0) {
                    clearInterval(interval);
                    stopWarning();
                    playFinal();
                    alert('TIME UP!');
                    resetTimer(flightIndex);
                }
            }, 100);
            
            card.dataset.intervalId = interval;
        }
        
        function stopTimer(flightIndex) {
            const card = document.getElementById(`flight-${flightIndex}`);
            clearInterval(card.dataset.intervalId);
            stopWarning();
            resetTimer(flightIndex);
        }
        
        function resetTimer(flightIndex) {
            const card = document.getElementById(`flight-${flightIndex}`);
            const timerDisplay = document.getElementById(`timer-${flightIndex}`);
            const startBtn = document.getElementById(`start-btn-${flightIndex}`);
            const stopBtn = document.getElementById(`stop-btn-${flightIndex}`);
            
            document.querySelectorAll('.flight-card').forEach(c => c.style.display = 'block');
            card.classList.remove('timer-active', 'timer-warning');
            timerDisplay.classList.remove('warning');
            timerDisplay.textContent = 'Ready to start';
            startBtn.style.display = 'block';
            stopBtn.style.display = 'none';
            activeTimer = null;
        }
        
        function playWarning() {
            try {
                stopWarning();
                warningOscillator = audioContext.createOscillator();
                warningGain = audioContext.createGain();
                warningOscillator.connect(warningGain);
                warningGain.connect(audioContext.destination);
                warningOscillator.frequency.value = 800;
                let time = audioContext.currentTime;
                for (let i = 0; i < 240; i++) {
                    warningGain.gain.setValueAtTime(0.3, time);
                    warningGain.gain.setValueAtTime(0, time + 0.2);
                    time += 0.5;
                }
                warningOscillator.start();
            } catch (e) {
                console.error('Warning sound error:', e);
            }
        }
        
        function stopWarning() {
            try {
                if (warningOscillator) {
                    warningOscillator.stop();
                    warningOscillator.disconnect();
                    warningOscillator = null;
                }
                if (warningGain) {
                    warningGain.disconnect();
                    warningGain = null;
                }
            } catch (e) {}
        }
        
        function playFinal() {
            try {
                const osc = audioContext.createOscillator();
                const gain = audioContext.createGain();
                osc.connect(gain);
                gain.connect(audioContext.destination);
                osc.frequency.value = 1000;
                gain.gain.setValueAtTime(0.5, audioContext.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 1.5);
                osc.start();
                osc.stop(audioContext.currentTime + 1.5);
            } catch (e) {}
        }
